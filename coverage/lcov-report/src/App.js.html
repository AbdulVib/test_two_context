import React, { Component } from 'react'
import { Form, FormControl, Button } from 'react-bootstrap'
import { bake_cookie, read_cookie, delete_cookie } from 'sfcookies'
import Note from './components/Note'
 
const cookie_key = 'NOTES'
 
export default class App extends Component {
  state = {
    text: '',
    notes: []
  }
 
  componentDidMount() {
    const notes = read_cookie(cookie_key)
 
    this.setState({ notes });
  }
 
  submit(){
    const { notes, text } = this.state
 
    notes.push({ text })
 
    this.setState({ notes, text: '' })
 
    bake_cookie(cookie_key, notes)
  }
 
  clear(){
    delete_cookie(cookie_key)
    this.setState({ notes: [] });
  }
 
  render() {
    return (
      <div className="container">
        <h2>Note to Self</h2>
 
        <Form>
          <FormControl onChange={ e => this.setState({ text: e.target.value })} value={ this.state.text } /> <br />
          <Button className="sbmt" onClick={ () => this.submit() }>Submit</Button>
        </Form>
 
      {
        this.state.notes.map((note, i) => <Note key={ i } note={ note } />)
      }
      <hr />
      <Button className="clear" onClick={() => this.clear() }>Clear Notes</Button>
      </div>
    )
  }
}
